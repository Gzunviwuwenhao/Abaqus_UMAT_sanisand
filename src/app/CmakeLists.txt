# project name
project(mymain)

# executable
add_executable(${PROJECT_NAME} main.f90)

# Add module directory to include path
target_include_directories(${PROJECT_NAME}
    PRIVATE ${CMAKE_Fortran_MODULE_DIRECTORY}
)

target_link_libraries(${PROJECT_NAME}
    PRIVATE
    myutils
)

# Don't compile module files directly, rely on myutils library
set_target_properties(${PROJECT_NAME} PROPERTIES
    Fortran_MODULE_DIRECTORY ""
)
